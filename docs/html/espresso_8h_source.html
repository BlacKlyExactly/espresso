476/**
477 * @brief Define a CORS middleware for handling preflight requests
478 *
479 * This macro generates a static function that sets the necessary
480 * CORS headers on the response and handles OPTIONS preflight requests.
481 *
482 * @param name Name of the generated function
483 * @param origin_val Value for "Access-Control-Allow-Origin" header
484 * @param methods_val Value for "Access-Control-Allow-Methods" header
485 * @param headers_val Value for "Access-Control-Allow-Headers" header
486 * @param max_age_val Value for "Access-Control-Max-Age" header (in seconds)
487 *
488 * @return Returns 1 if the request is an OPTIONS preflight request and
489 * the response was sent, otherwise 0.
490 *
491 * @note The generated function can be used as a middleware in espresso:
492 * ```
493 * DEFINE_CORS(cors_allow_all, "*", "GET, POST, PATCH, DELETE, OPTIONS",
494 * "Content-Type, Authorization", 86400)
495 * app_use(app, cors_allow_all);
496 * ```
497 */
498#define DEFINE_CORS(name, origin_val, methods_val, headers_val, max_age_val) \
499 static int name(ResponseContext *res) { \
500 set_header(res, "Access-Control-Allow-Origin", origin_val); \
501 set_header(res, "Access-Control-Allow-Methods", methods_val); \
502 set_header(res, "Access-Control-Allow-Headers", headers_val); \
503 char max_age[32]; \
504 snprintf(max_age, sizeof(max_age), "%d", max_age_val); \
505 set_header(res, "Access-Control-Max-Age", max_age); \
506 if (strcmp(res->req->method, "OPTIONS") == 0) { \
507 res->status = 204; \
508 send_text_response(res, ""); \
509 return 1; \
510 } \
511 return 0; \
512 }
